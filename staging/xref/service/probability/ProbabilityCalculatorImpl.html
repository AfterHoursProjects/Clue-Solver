33  		Probability<Weapon> mostLikelyWeapon = weapons.get(0);

38  
39  		double tripleProbability = mostLikelyRoom.getCardProbability() * mostLikelySuspect.getCardProbability()
40  				* mostLikelyWeapon.getCardProbability();

42  				.getWrappedObject(), mostLikelyWeapon.getWrappedObject()), tripleProbability));

56  	private <F extends Card> List<Probability<F>> toSortedProbabilities(Iterable<ObjectCounter<F>> toConvert) {
57  		List<Probability<F>> converted = Lists.newArrayList();
58  		double size = 0;
59  
60  		// Has to loop twice first to get total instances in that list, second to set the proabilities
61  		for (ObjectCounter<F> counter : toConvert) {
62  			size += counter.getCount();
63  		}
64  
65  		for (ObjectCounter<F> counter : toConvert) {
66  			converted.add(new Probability<F>(counter.getCard(), counter.getCount() / size));
67  		}

69  		Collections.sort(converted);
