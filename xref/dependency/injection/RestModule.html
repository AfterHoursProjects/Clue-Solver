32  public class RestModule extends AbstractModule {

34  	@Override
35  	protected void configure() {

37  
38  		// This binding is used to set the port number, could be modified to something that
39  		// read an external config file, command line argument, jvm argument, etc
40  		bind(Integer.class).annotatedWith(Names.named("serverPort")).toInstance(1234);
41  
42  		// Arguments for the thread pool setup
43  		bind(Integer.class).annotatedWith(Names.named("coreThreadPoolSize")).toInstance(10);
44  		bind(Integer.class).annotatedWith(Names.named("maxThreads")).toInstance(30);
45  		bind(Integer.class).annotatedWith(Names.named("extraThreadsTimeout")).toInstance(10);
46  
47  		// Creates the blocking queue for jobs waiting to execute
48  		BlockingQueue<Runnable> blockingQueue = Queues.newLinkedBlockingQueue(500);
49  		bind(new TypeLiteral<BlockingQueue<Runnable>>() {
50  		}).annotatedWith(Names.named("threadQueue")).toInstance(blockingQueue);
51  
52  		// Setup the executor service which should manage all application threads
53  		bind(ExecutorService.class).toProvider(ExecutorServiceProvider.class).in(Scopes.SINGLETON);
54  
55  		// Setup the task executor service, this wraps up the executor service into the restlet task service interface
56  		bind(TaskService.class).to(ClueTaskService.class).in(Scopes.SINGLETON);
57  
58  		// Binds the ClueServer class to the ClueServer builder
59  		bind(ClueServer.class).toProvider(ClueServerProvider.class).in(Scopes.SINGLETON);
60  
61  		// Binds the user input listener
62  		bind(InputListener.class).to(UserInputListener.class).in(Scopes.SINGLETON);
63  
64  		// Binds the console controller
